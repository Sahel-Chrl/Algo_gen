Mon premier algorithme génétique : AG-parabole3 est un algorithme qui a pour objectif de prendre des points au hasard dans R2
et de trouver les polynômes d'un degré fixé pour lesquels la distance moyenne (verticale) des points à la courbe est la plus
faible possible.


Premièrement l'algo des points qu'il répartira aléatoirement dans une fenêtre. Ces points correspondront
à l'environnement auquel les individus devront s'adapter. Pour nous les humains on doit s'adapter le mieux possible
à la météo autour de nous ou bien aux prédateurs, eh bien la météo peut être assimilée à un de ces points et les
prédateurs à un autre.

Les polynômes seront donc les individus qui évoluent pour s'adapter au mieux possible à l'environnement c.-à-d. ici
se rapprocher le plus possible des points. Dans l'exemple juste au dessus les humains sont donc les polynômes.

Une population est constituée de n paraboles qui sont " en concurrence " celle qui est la plus proche des points
aura plus de chances de se reproduire, la plus éloignée n'aura surement aucun descendant.

Lorsqu'un parent est choisi il est créé un descendant qui aura des coefficients proches de celui du parent, on
ajoute à chaque coefficient une mutation qui fait en sorte que les descendants ne soient pas un clone de son parent.
Ces mutation permettront de se rapprocher des points au fur et a mesure des générations.
On ajoute également une probabilité de grosse mutation qui quand elle a lieu donne un descendant bien plus éloigné
des autres, cette methode permet d'éviter de stagner dans des minimums locaux lorsqu'en réalité il existe une
meilleure option.
Dans l'exemple précédent on pourrait assimiler les coefficients des polynômes aux gènes des individus.


Avant de commencer : le code vous demandera 4 choses
- le nombre de points que vous voulez, je vous conseille un nombre entre 4 et 8 pour qu'il ne soit pas trop compliqué
pour les courbes de se rapprocher des points.
- le nombre de coefficients par polynôme, je vous conseille 10. Ce nombre de coefficients laisse plus ou moins de
liberté a nos polynômes, si vous mettez deux vous aurez des droites.
- le nombre d'individus par population, je vous conseille une trentaine.
- le nombre de générations, je vous conseille 10000, plus il y a de générations plus on se rapprochera des points, en
mettant beaucoup de générations vous pourrez juste décider du moment où vous arrêter en fermant la fenêtre.

Pour tester l'efficacité du programme vous pouvez essayer avec 2 points, 2 coefficients et 5 courbes, normalement les
droites atteindront rapidement la perfection.

class Point {
  public float x;
  public float y;
  Point(float i, float j) {
    x=i;
    y=j;
  }
}

class Parabole
{
  float[] coefs;
  color couleur;

  public Parabole(int degre, float[] grandeurs)
  { 
    coefs = new float[degre];
    for (int j = 0; j < degre; j++)
    {
      float x;
      x = hasard1Neg() * grandeurs[j];
      coefs[j] = x;
    }
    couleur = random_col();
  }

  Parabole( float[] coefficients, color col)
  {
    coefs = new float[coefficients.length];
    for (int i=0; i<coefficients.length; i++)
      coefs[i]=coefficients[i];
      
    couleur = col;
  }
};

import javax.swing.JOptionPane;


float hasard()
{
  return (float)random(1000);
} // fonction nombre aleatoire pour mes parab

float hasard1()
{ 
  return (float)random(1);
}

float hasard1Neg()
{
  return (float)random(-1,1);
}

color random_col()
{
  return color(
    hasard1() * 255,
    hasard1() * 255,
    hasard1() * 255);
}

int lireInt(String message) {

// fonction qui crée les couples de points
Point[] couple(int n)
{
  Point[] points = new Point[n];
  
  for (int i = 0; i < n; i++)
  {
    float x = hasard1Neg() * 100;
    float y = hasard1Neg() * 100;
    points[i]= new Point(x, y); // on range le couple cree
  }
  return points;
}

// on va creer ici notre popuulation de parabs
Parabole[] create_parab(int n, int degre, float[] grandeurs)
{
  Parabole[] result = new Parabole[n];

  for (int i = 0; i < n; i++) // je parcours les courbes
  {
    //Parabole par = new Parabole(degre, grandeurs);
    //result.push(par);
    result[i] = new Parabole(degre, grandeurs);
  }
  return result;
}

// fonction qui calcule la valeur en x
float valeur(float x, Parabole p)
{
  float y = 0;
  // y = p[0] * x * x + p[1] * x + p[2];
  for (int i = 0; i < p.coefs.length; i++)
  {
    y = y * x + p.coefs[i];
  }
  return y;
}

// fonction distance verticale
float distance(float abcisse, float ordonnee, Parabole parab)
{
  float y, d;
  y = valeur(abcisse, parab);
  d = abs(y - ordonnee);
  return d;
}

// fonction distance moyenne
float distancemoy(Parabole c, Point[] Point)
{
  float d;
  int n = Point.length;
  d = 0;
  for (var point : Point)
  {
    d = d + distance(point.x, point.y, c);
  }
  d = d / n;
  return d;
}

// fonction qui prends les scores des parabs et les transforme en probas
float[] score2proba(float[] scores)
{
  float[] probas = new float[scores.length];
  float somme = 0;
  
  for ( int i=0; i<scores.length; i++){ // plus le score est grand moins il aura de chance d'être pris (on veut la distance min)
  float inverse = 1/scores[i];
  probas[i]=inverse;
  somme+=inverse;
  }
  
  if (somme == 0)
  {
    println ( "attention somme nulle" );
    return probas;
  }

  if (somme==0){
    println("attention somme nulle division par 0 dans score2probas");
  }
  // on normalise notre tableaux de probas pour que la somme fasse 1
  for (int j = 0; j < probas.length; j++)
  {
    probas[j] = probas[j] / somme;
  }
  
  return probas;
}

// on fait juste la somme des probas pour les
float[] probas2cumul(float[] probas)
{
  float[] cumul = new float[probas.length+1];
  cumul[0]=0;
  for (int i = 0; i < probas.length ; i++)
  {
    cumul[i+1]=probas[i]+cumul[i];
  }
  return cumul;
}

// fonction qui prend au hasard un nombre entre 0 et 1 et revoie la position de ce nombre dans cumul (renvoi quel individu c'est)
int echantillon(float[] cumul)
{
  float r = hasard1();
  int debut = 0, fin = cumul.length - 1;
  while (fin - debut > 1)
  {
    int milieu = (fin + debut) / 2;
    if (cumul[milieu] > r)
    {
      fin = milieu;
    } else
    {
      debut = milieu;
    }
  }
  return debut;
}

// fonction qui crée un descendant en fonction d'un parent choisi, c'est surement ici qu'on peut encore ameliorer le code
Parabole descendant(Parabole mere, float variance, float[] grandeurs)
{
  float[] enfantCoefs = new float[mere.coefs.length];
  float newVar = variance;
  color newCol = mere.couleur;
  if (hasard1() < .02)
  {
    newVar = 40 * variance; // grosse mutation
    newCol = random_col();
  }

  for (int i = 0; i < mere.coefs.length; i++)
  {
    
    float hasard = hasard1Neg();

    enfantCoefs[i] = mere.coefs[i] + newVar * hasard * grandeurs[i]; // enfant + mutation
  }
  
  Parabole enfant = new Parabole(enfantCoefs, newCol);
  return enfant;
  
} // je pourrais changer cette fonction pour moins changer le x carre que la  ante ?

// fonction qui prend les scores d'une population et donne la distance minimale parmis ces scores
float minVec(float[] scores)
{
  if (scores.length == 0)
  {
    println( "scores vide, erreur" );
    return 0.;
  }
  float min = scores[0];
  for (int i = 0; i < scores.length; i++)
  {
    if (scores[i] < min)
    {
      min = scores[i];
    }
  }
  return min;
}

// fonction qui va calculer le score moyen pour ensuie avoir une meilleur estimation de la variance
float scoreMoyen(float[] scores)
{
  float moy;
  moy = 0;
  for (int i = 0; i < scores.length; i++)
  {
    moy = moy + scores[i];
  }
  moy = moy / scores.length;
  return moy;
}




int nb_checkpoint = 100;

int n, m, deg, generations;
int idGeneration = 0;
int numPopu = 0;

Point[] points;
float[] grandeurs;
Parabole[] popu;

int pointCount = 200;
float radius = 3;

void setup() {
  size(800, 800);
  println("setup debut");
  randomSeed((int)millis());

  n = lireInt( "combien de points voulez vous ?");
  points = couple(n);

  deg = lireInt("combien de coefs veux-tu pour les polynomes ?");
  if (deg < 1) deg = 1;

  grandeurs = new float[deg];
  float g = 100;
  for (int d = 0; d < deg; d++)
  {
    grandeurs[deg - d - 1] = g;
    g = g / 100;
  }


  m = lireInt("combien de polynomes voulez vous ?");
  if (m < 1) m = 1;
  // appel d'une liste de paraboles pour la population
  popu = create_parab(m, deg, grandeurs);


  float[] scores = new float[popu.length];
  for ( int i=0; i<popu.length; i++)
  {
    scores[i]= distancemoy(popu[i], points);
  }

  generations = lireInt("combien de generations voulez vous ?");
  if (generations < 1 ) generations = 1;
  println("setup fin");
}


void dessinerAxes() {
  stroke(150);
  line(0, 400, 800, 400);
  line(400, 0, 400, 800);
}

void dessinerPoints() {
  fill(255, 0, 0);
  noStroke();

  for (Point point : points) {
    float px = (point.x +100)*4;
    float py = (point.y +100)*4;
    ellipse(px, py, radius*2, radius*2);
  }
}


void dessinerCourbes() {
  noFill();

  for (Parabole par : popu) {
    stroke(par.couleur);

    beginShape();
    for (int i = 0; i < pointCount; i++) {
      float x = i - 100;
      float y = valeur(x, par);

      float px = (x + 100) * 4;
      float py = (y + 100) * 4;

      vertex(px, py);
    }
    endShape();
  }
}


void draw() {
  background(0);

  dessinerAxes();
  dessinerPoints();

  if (idGeneration < generations) {

    int distance_Checkpoint = max(1, generations / nb_checkpoint);

    Parabole[] newPop = new Parabole[popu.length];
    float[] scores = new float[popu.length];

    for (int i = 0; i < popu.length; i++) {
      scores[i] = distancemoy(popu[i], points);
    }

    float[] probas = score2proba(scores);
    float[] cumul = probas2cumul(probas);

    for (int j = 0; j < popu.length; j++) {
      int indiv = echantillon(cumul);
      float variance = 0.05;

      if (indiv >= popu.length) {
        println("erreur: indiv " + indiv + " dans popu " + popu.length);
        indiv = popu.length - 1;
      }

      Parabole enfant = descendant(popu[indiv], variance, grandeurs);
      newPop[j] = enfant;
    }

    popu = newPop;

    if (idGeneration % distance_Checkpoint == 0) {
      println("Population numero " + numPopu + " : ");
      numPopu++;
      println(minVec(scores) + " " + scoreMoyen(scores));
    }

    idGeneration++;
  }

  dessinerCourbes();
}
